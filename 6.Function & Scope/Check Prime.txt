import java.util.Scanner;
public class CheckPrime {
public static boolean checkPrime(int n){
    int div=2;
    
    while(div<=n/2){
        if(n%div==0){
            //Not Prime
            return false;
        }
        div=div+1;
    }
    return true;
}
